let n1 : number = prompt("Inserisci il primo numero")*1;
let n2 : number = prompt("Inserisci il secondo numero")*1;
let somma : number = 0;
let sottrazione : number = 0;
let moltiplicazione : number = 0;
let divisione : number = 0;
somma = n1 + n2 alert ("La sommma è " + somma);
sottrazione = n1 - n2 alert ("La sottrazione è " + sottrazione);
moltiplicazione = n1 * n2 alert ("La moltiplicazione è " + moltiplicazione);
divisione = n1 / n2 alert("La divisione è " + divisione);
